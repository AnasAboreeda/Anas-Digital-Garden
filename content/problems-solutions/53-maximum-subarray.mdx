# 53 Maximum Subarray

## Problem Statement

## Solution

```java

class Solution {
    private int slowMaxSubArray(int[] nums) {
        int maxSum = nums[0];
        
        for (int i = 0; i < nums.length; i++) {
            for(int j = i; j < nums.length; j++) {
                int sum = sumBet(nums, i, j);
                if(sum > maxSum) {
                    maxSum = sum;
                }
            }
        }
        
        return maxSum;
    }
    
    private int sumBet(int[] nums, int i, int j) {
        return Arrays.stream(nums).skip(i).limit(j - i + 1).reduce(0, Integer::sum);
    }
    
    
    private int quickMaxSubArray(int[] nums) {
       int currentSum = nums[0];
        int globalSum = nums[0];
        
        for (int i = 1; i < nums.length; i++) {
            currentSum = Math.max(nums[i], currentSum + nums[i]);
            if(globalSum < currentSum) {
                globalSum = currentSum;
            }      
        }
      
        
        return globalSum;
    }
    
    
    public int maxSubArray(int[] nums) {
        return quickMaxSubArray(nums);
    }
}
```
